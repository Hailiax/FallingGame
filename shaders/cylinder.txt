uniform float time;
uniform vec2 resolution;

varying vec3 vPosition;
varying vec2 vUv;
  
void main()	{
    float value = fbm(vec4(vPosition.x * 2., vPosition.y - time/50., vPosition.z * 2. , 0.));
    value += 1.;
    value /= 2.;
    value *= value;
    gl_FragColor = vec4(value, value, value, value * 2.);
}
